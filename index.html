//Nhap SV tu stream (chung ta se dung stream la file)

void readSV(istream& in, SV &x)

{

in>>x.ten;

in>>x.tuoi;

in>>x.gioi;

in>>x.dc;

in>>x.diem;

}

//Hien thi SV ra dong (chung ta se dung stream la file)

void printSV(ostream& out, SV x)

{

out<<x.ten<<" "<<x.tuoi<<" "<<x.gioi<<" "<<x.dc<<" "<<x.diem<<endl;

}

//Doc mang tu file

void readMang(char fileName[], MangSV &x)

{

ifstream fin(fileName);

int i=0;

while(!fin.eof())

{

readSV(fin,x.a[i]);

i++;

}

x.n=i-1;

fin.close();

}

//Ghi mang ra file

void printMang(char fileName[], MangSV x)

{

ofstream fout(fileName);

for(int i=0;i<x.n;i++)

{

printSV(fout,x.a[i]);

}

fout.close();

}

void HoanVi(SV& a,SV& b)

{

SV t=a;a=b;b=t;

}

//cap nhat ten phan tu i

void CapNhatTen(string ten,int i, MangSV &x)

{

if (i>=x.n || i<0)return;

x.a[i].ten=ten;

}

//cap nhat diem phan tu i

void CapNhatDiem(int m,int i, MangSV &x)

{

if (i>=x.n || i<0)return;

x.a[i].diem=m;

}

//sap xep danh sach theo diem tang

void SapXepDiemTang(MangSV &x)

{

int i,j,max;

for(i=0;i<x.n-1;i++)

{

max=i;

for(j=i+1;j<x.n;j++)

if(x.a[j].diem<x.a[max].diem)

HoanVi(x.a[max],x.a[j]);

}

}

 

//so luong sinh vien nam

int SoLuongNam(MangSV x)

{

int i,nam=0;

for(i=0;i<x.n;i++)

if(x.a[i].gioi=='n')nam++;

return nam;

}
